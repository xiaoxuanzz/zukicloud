cURL 调用示例

1. 上传文件

# 上传文件到根目录
curl -X POST "http://your-domain.com/api/index.php?action=upload" \
  -H "X-API-KEY: your_api_key_here" \
  -F "file=@/path/to/your/file.jpg"

# 上传文件到指定目录
curl -X POST "http://your-domain.com/api/index.php?action=upload" \
  -H "X-API-KEY: your_api_key_here" \
  -F "file=@/path/to/your/file.jpg" \
  -F "dest=images/2026"

2. 获取文件列表

# 获取根目录文件列表
curl -H "X-API-KEY: your_api_key_here" \
  "http://your-domain.com/api/index.php?action=list"

# 递归获取所有文件
curl -H "X-API-KEY: your_api_key_here" \
  "http://your-domain.com/api/index.php?action=list&recursive=true"

3. 下载文件

# 下载文件（使用hash）
curl -H "X-API-KEY: your_api_key_here" \
  -O -J \
  "http://your-domain.com/api/index.php?action=file&path=file_hash_here"

JavaScript 调用示例

1. 上传文件

// 上传文件
async function uploadFile(file) {
    const formData = new FormData();
    formData.append('file', file);
    formData.append('dest', 'images'); // 可选，目标目录
    
    try {
        const response = await fetch('http://your-domain.com/api/index.php?action=upload', {
            method: 'POST',
            headers: {
                'X-API-KEY': 'your_api_key_here'
            },
            body: formData
        });
        const data = await response.json();
        if (data.success) {
            console.log('上传成功:', data);
        }
    } catch (error) {
        console.error('上传失败:', error);
    }
}

2. 获取文件列表

// 获取文件列表
async function loadFileList() {
    try {
        const response = await fetch(
            'http://your-domain.com/api/index.php?action=list',
            {
                headers: {
                    'X-API-KEY': 'your_api_key_here'
                }
            }
        );
        const data = await response.json();
        if (data.contents) {
            console.log('文件列表:', data.contents);
        }
    } catch (error) {
        console.error('获取失败:', error);
    }
}

Python 调用示例

1. 上传文件

import requests

# 上传文件
url = "http://your-domain.com/api/index.php?action=upload"
headers = {"X-API-KEY": "your_api_key_here"}
files = {"file": open("file.jpg", "rb")}
data = {"dest": "images"}  # 可选

response = requests.post(url, headers=headers, files=files, data=data)
print(response.json())

2. 获取文件列表

import requests

# 获取文件列表
url = "http://your-domain.com/api/index.php?action=list"
headers = {"X-API-KEY": "your_api_key_here"}

response = requests.get(url, headers=headers)
print(response.json())

注意：请将示例中的域名和API密钥替换为你自己的。
